	curr_min = a[0]
	for(i = 1; i < n ; i++) {
		if(a[i] <= curr_min) {
			curr_min = a[i];
		}else {
			curr_max_diff = max(curr_max_diff, a[i] - curr_min);
		}
	}

				if ele > back (remove back recursively till ele < back)
		front of aux queue always gives current max. Similarly for min.
